//EXEMPLO DE CASESENSITIVE, INFORMANDO TANTO MAIUSCULAS COMO MINUSCULAS COM TOLOWERCASE
let recebeMes = prompt("Digite o mês:")
let mes = recebeMes.toLowerCase()

switch (mes) {
    case 'dezembro':
        console.log("Dezembro")
        break;
    case 'janeiro':
        console.log("Janeiro")
        break;
    default:
        console.log("Mês invalido")
        break;
}
